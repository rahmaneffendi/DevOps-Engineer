kemudian saya akan membuat ansible playbook untuk docker

```
- hosts: all
  become: yes
  gather_facts: yes
  tasks:
         - name: 'update'
           apt:
            update_cache: yes

         - name: 'upgrade'
           apt:
            upgrade: dist


         - name: 'install dependencies'
           apt:
             name:
             - ca-certificates
             - curl
             - gnupg
             - python3-pip
             - lsb-release

         - name: 'add docker gpg key'
           apt_key:
            url: https://download.docker.com/linux/ubuntu/gpg

         - name: 'add repository docker'
           apt_repository:
             repo: deb  https://download.docker.com/linux/ubuntu focal stable

         - name: 'install docker engine'
           apt: 
            name:
             - docker-ce
             - docker-ce-cli
             - containerd.io
             - docker-compose-plugin

         - name: 'update'
           apt:
            update_cache: yes

         - name: 'install docker-compose'
           shell: curl -SL https://github.com/docker/compose/releases/download/v2.6.0/docker-compose-linux-x86_64 -o /usr/local/bin/docker-compose


         - name: 'set permision for docker'
           shell: sudo chmod +x /usr/local/bin/docker-compose
           
- hosts: app
  become: yes
  gather_facts: yes
  tasks:
     
         - name: 'docker without sudo'
           shell: sudo usermod -aG docker app
           
- hosts: gateway
  become: yes
  gather_facts: yes
  tasks:
     
         - name: 'docker without sudo'
           shell: sudo usermod -aG docker nginx
  
- hosts: monitoring
  become: yes
  gather_facts: yes
  tasks:
     
         - name: 'docker without sudo'
           shell: sudo usermod -aG docker moni
           
- hosts: jenkins
  become: yes
  gather_facts: yes
  tasks:
     
         - name: 'docker without sudo'
           shell: sudo usermod -aG docker jenkins
           
```
