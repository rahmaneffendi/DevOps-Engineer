Selanjutnya membuat Docker dan docker-compose.yml untuk backend 

```
FROM node:dubnium-alpine3.11
WORKDIR /usr/app
COPY . .
RUN npm install
RUN npm run build
RUN npm install sequelize-cli -g
RUN npx sequelize db:migrate
EXPOSE 5000
CMD [ "node", "server.js" ]

```
